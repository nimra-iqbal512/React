===== Create and use store
== To install redux toolkit:
npm install @reduxjs/toolkit react-redux

== src\store.js:
import { configureStore } from "@reduxjs/toolkit";

const store = configureStore({
    reducer: {},
})

export default store;


== Reducers.js:
import { createReducer } from "@reduxjs/toolkit";

// // createReducer has two objects
// export const customReducer = createReducer({}, {})

// Defining objects separately
const initialState = {
  c: 0,
};
export const customReducer = createReducer(initialState, {
  increment: (state, action) => {
    state.c += 1;
  },
  incrementByValue: (state, action) => {
    state.c += action.payload; //payload is the parameter passed by user
  },
  decrement: (state, action) => {
    state.c -= 1;
  },
});

import { createReducer } from "@reduxjs/toolkit";

// // createReducer has two objects
// export const customReducer = createReducer({}, {})

// // Defining objects separately, (but this has been depreciated)
// const initialState = {
//   c: 0,
// };
// export const customReducer = createReducer(initialState, {
//   increment: (state, action) => {
//     state.c += 1;
//   },
//   incrementByValue: (state, action) => {
//     state.c += action.payload; //payload is the parameter passed by user
//   },
//   decrement: (state, action) => {
//     state.c -= 1;
//   },
// });

// ==============================
// ✅ Old (❌ Deprecated):

// js
// Copy code
// createReducer(initialState, {
//   increment: (state) => { ... }
// })
// ✅ New (✔️ Recommended):

// js
// Copy code
// createReducer(initialState, (builder) => {
//   builder.addCase("increment", (state) => { ... })
// })
// ==================================

const initialState = {
  c: 0,
};
export const customReducer = createReducer(initialState, (builder) => {
  builder
    .addCase("increment", (state) => {
      state.c += 1;
    })
    .addCase("incrementByValue", (state, action) => {
      state.c += action.payload; //payload is the parameter passed by user
    })
    .addCase("decrement", (state) => {
      state.c -= 1;
    });
});

== Home.js:
import { useDispatch } from "react-redux";
// useDispatch is to trigger action
// useSelector is to access state value

const Home = () => {
  const value = 20;
  const dispatch = useDispatch();

  const addBtn = () => {
    dispatch({
      type: "increment",
    });
  };
  const addBtn25 = () => {
    dispatch({
      type: "incrementByValue",
      payload: 25,
    });
  };
  const subBtn = () => {
    dispatch({
      type: "decrement",
    });
  };
  return (
    <div>
      <h2>{value}</h2>

      <button onClick={addBtn}>Increment</button>
      <button onClick={addBtn25}>Increment by 25</button>
      <button onClick={subBtn}>Decrement</button>
    </div>
  );
};
export default Home;
